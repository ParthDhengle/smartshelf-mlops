CREATE TABLE IF NOT EXISTS public.weather
(
    weather_id integer NOT NULL,
    store_id integer,
    weather_date date,
    temperature_c numeric,
    rainfall_mm numeric,
    humidity_pct numeric,
    weather_type character varying(50) COLLATE pg_catalog."default",
    CONSTRAINT weather_pkey PRIMARY KEY (weather_id)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.weather
    OWNER to postgres;
-- Index: idx_weather_store_date

-- DROP INDEX IF EXISTS public.idx_weather_store_date;

CREATE INDEX IF NOT EXISTS idx_weather_store_date
    ON public.weather USING btree
    (store_id ASC NULLS LAST, weather_date ASC NULLS LAST)
    WITH (fillfactor=100, deduplicate_items=True)
    TABLESPACE pg_default;
